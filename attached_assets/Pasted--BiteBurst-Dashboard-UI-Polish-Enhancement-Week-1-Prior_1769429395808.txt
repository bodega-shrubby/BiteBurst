### **Task 4: Add XP Burst Animation**

**File:** `client/src/pages/DashboardV2.tsx`

**What to do:**
1. Create an XP burst animation component that shows "+10 XP", "+20 XP", etc.
2. Trigger it when user completes Daily Journey tasks or when XP updates
3. Animation should:
   - Float upward and fade out
   - Have orange color (#FF6A00)
   - Show for 1.5 seconds then disappear

**Create XPBurst component:**
```tsx
interface XPBurstProps {
  amount: number;
  visible: boolean;
  onComplete: () => void;
}

function XPBurst({ amount, visible, onComplete }: XPBurstProps) {
  useEffect(() => {
    if (visible) {
      const timer = setTimeout(onComplete, 1500);
      return () => clearTimeout(timer);
    }
  }, [visible, onComplete]);

  if (!visible) return null;

  return (
    <div className="fixed top-1/2 left-1/2 transform -translate-x-1/2 -translate-y-1/2 z-50 pointer-events-none">
      <div className="xp-burst-animation">
        <span className="text-5xl font-black text-orange-500 drop-shadow-lg">
          +{amount} XP
        </span>
      </div>
    </div>
  );
}
```

**Add CSS animation:**
```css
@keyframes xp-burst {
  0% {
    transform: scale(0.5) translateY(0);
    opacity: 0;
  }
  20% {
    transform: scale(1.3) translateY(-20px);
    opacity: 1;
  }
  100% {
    transform: scale(1) translateY(-80px);
    opacity: 0;
  }
}

.xp-burst-animation {
  animation: xp-burst 1.5s ease-out forwards;
}
```

**Add state management:**
```tsx

  amount: 0,

  setXpBurst({ amount: xpReward, visible: true });
};

// In JSX
<XPBurst

  visible={xpBurst.visible}
  onComplete={() => setXpBurst({ ...xpBurst, visible: false })}
```

**Improve TodaysJourney component:**
```tsx
interface TodaysJourneyProps {
  milestones: Array<{
    id: string;
    title: string;
    reward: number;
    completed: boolean;
  }>;
  onTaskComplete?: (xpReward: number) => void;
}

function TodaysJourney({ milestones, onTaskComplete }: TodaysJourneyProps) {
  const completedCount = milestones.filter(m => m.completed).length;
  const allComplete = completedCount === milestones.length;
  const progress = (completedCount / milestones.length) * 100;

  // Trigger celebration when all tasks complete
  useEffect(() => {
    if (allComplete && onTaskComplete) {
      onTaskComplete(50); // Daily Journey bonus
    }
  }, [allComplete]);

  return (
    <div className="bg-white border-2 border-gray-200 rounded-2xl p-6 space-y-4">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-xl font-bold text-gray-900">Today's Journey</h3>
        <div className="text-sm font-semibold text-gray-600">
          {completedCount}/{milestones.length} tasks
        </div>
      </div>

      {/* Progress Bar */}
      <div className="h-3 bg-gray-100 rounded-full overflow-hidden">
        <div 
          className="h-full bg-gradient-to-r from-orange-400 to-orange-600 transition-all duration-500 ease-out"
          style={{ width: `${progress}%` }}
        />
      </div>

      {/* Tasks */}
      <div className="space-y-3">
        {milestones.map((milestone) => (
          <div 
            key={milestone.id}
            className={`
              flex items-center justify-between p-3 rounded-xl transition-all duration-300
              ${milestone.completed 
                ? 'bg-green-50 border-2 border-green-200' 
                : 'bg-gray-50 border-2 border-gray-200'
              }
            `}
          >
            <div className="flex items-center space-x-3">
              {/* Checkbox */}
              <div className={`
                w-6 h-6 rounded-full border-2 flex items-center justify-center transition-all duration-300
                ${milestone.completed 
                  ? 'bg-green-500 border-green-500 scale-110' 
                  : 'border-gray-300'
                }
              `}>
                {milestone.completed && (
                  <svg className="w-4 h-4 text-white" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M5 13l4 4L19 7" />
                  </svg>
                )}
              </div>

              <span className={`
                font-medium transition-all duration-300
                ${milestone.completed ? 'text-green-800 line-through' : 'text-gray-700'}
              `}>
                {milestone.title}
              </span>
            </div>

            <span className="text-orange-600 font-bold text-sm">
              +{milestone.reward} XP
            </span>
          </div>
        ))}
      </div>

      {/* Completion Bonus */}
      {allComplete && (
        <div className="mt-4 p-4 bg-gradient-to-r from-orange-100 to-yellow-100 border-2 border-orange-300 rounded-xl text-center animate-pulse">
          <p className="text-lg font-bold text-orange-800">
            üéâ Daily Journey Complete! +50 XP Bonus!
          </p>

      )}
    </div>
  );
}
```

```tsx
interface Log {
  id: string;
  type: 'food' | 'activity';
  summary: string;
  ts: string;
  xpAwarded?: number;
  feedback?: string;
}

interface RecentLogsListProps {
  logs: Log[];
}

function RecentLogsList({ logs }: RecentLogsListProps) {
  const getTimeAgo = (timestamp: string) => {
    const now = new Date();
    const logTime = new Date(timestamp);
    const diffMs = now.getTime() - logTime.getTime();
    const diffMins = Math.floor(diffMs / 60000);
    
    if (diffMins < 1) return 'Just now';
    if (diffMins < 60) return `${diffMins} min ago`;
    const diffHours = Math.floor(diffMins / 60);
    if (diffHours < 24) return `${diffHours} hour${diffHours > 1 ? 's' : ''} ago`;
    return 'Earlier';
  };

  const getIcon = (type: string) => {
    return type === 'food' ? 'üçΩÔ∏è' : 'üèÉ';
  };

  if (!logs || logs.length === 0) {
    return (
      <div className="bg-gradient-to-br from-orange-50 to-yellow-50 border-2 border-orange-200 rounded-2xl p-8 text-center">
        <div className="text-6xl mb-3">üìù</div>
        <h3 className="text-lg font-bold text-gray-900 mb-2">
          No activity yet today
        </h3>
        <p className="text-gray-600 text-sm mb-4">
          Tap the buttons below to log your first meal or activity!
        </p>
        <div className="flex justify-center">
          <img 
            src={sunnySliceImage} 
            alt="Sunny Slice" 
            className="w-20 h-20 animate-subtle-bounce"
          />
        </div>
      </div>
    );
  }

  return (
    <div className="space-y-4">
      <h2 className="text-xl font-bold text-gray-900">Recent Activity</h2>
      
      <div className="space-y-3">
        {logs.slice(0, 5).map((log) => (
          <div 
            key={log.id}
            className="bg-white border-2 border-gray-200 rounded-xl p-4 hover:border-orange-200 hover:shadow-md transition-all duration-200"
          >
            <div className="flex items-start justify-between mb-2">
              <div className="flex items-center space-x-2">
                <span className="text-2xl">{getIcon(log.type)}</span>
                <div>
                  <p className="font-semibold text-gray-900">{log.summary}</p>
                  <p className="text-xs text-gray-500">{getTimeAgo(log.ts)}</p>
                </div>
              </div>
              
              {log.xpAwarded && (
                <div className="bg-orange-100 px-3 py-1 rounded-full">
                  <span className="text-orange-600 font-bold text-sm">
                    +{log.xpAwarded} XP
                  </span>
                </div>
              )}
            </div>

            {/* AI Feedback Preview */}
            {log.feedback && (
              <div className="mt-2 p-3 bg-blue-50 rounded-lg border border-blue-100">
                <p className="text-sm text-gray-700">
                  üí¨ {log.feedback.substring(0, 100)}
                  {log.feedback.length > 100 && '...'}
                </p>
              </div>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```
