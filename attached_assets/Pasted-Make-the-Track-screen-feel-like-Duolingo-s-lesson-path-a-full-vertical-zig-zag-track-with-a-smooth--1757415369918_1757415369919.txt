Implement a dedicated PathCanvas that receives node anchor points and draws a smooth centerline:
type Point = { x: number; y: number };

function makeSmoothPath(points: Point[]): string {
  // Use cubic Beziers between midpoints for smoothness
  if (points.length < 2) return '';
  const d = [`M ${points[0].x} ${points[0].y}`];
  for (let i = 1; i < points.length; i++) {
    const p0 = points[i - 1];
    const p1 = points[i];
    const cx = (p0.x + p1.x) / 2;
    d.push(`C ${cx} ${p0.y}, ${cx} ${p1.y}, ${p1.x} ${p1.y}`);
  }
  return d.join(' ');
}
