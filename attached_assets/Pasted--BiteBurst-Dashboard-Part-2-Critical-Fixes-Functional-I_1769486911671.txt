### **Task 1: XP Burst Animation ‚ö°**

**File:** `client/src/pages/DashboardV2.tsx`

**Problem:** When users log food/activity and return to dashboard, there's no visual celebration of XP earned.

**Solution:** Create a floating "+20 XP" animation that appears when user returns from logging.

**Step 1: Create XPBurst Component**
```tsx
// Add this component near the top of DashboardV2.tsx after imports
interface XPBurstProps {
  amount: number;
  visible: boolean;
  onComplete: () => void;
}

function XPBurst({ amount, visible, onComplete }: XPBurstProps) {
  useEffect(() => {
    if (visible) {
      const timer = setTimeout(onComplete, 2000); // Show for 2 seconds
      return () => clearTimeout(timer);
    }
  }, [visible, onComplete]);

  if (!visible) return null;

  return (
    <div className="fixed inset-0 pointer-events-none z-50 flex items-center justify-center">
      <div className="xp-burst-animation">
        <div className="text-6xl font-black text-orange-500 drop-shadow-2xl">
          +{amount} XP
        </div>
      </div>
    </div>
  );
}
```

### **Task 2: Daily Journey Auto-Complete ‚úÖ**

**File:** `client/src/components/dashboard/TodaysJourney.tsx`

**Problem:** Daily Journey tasks don't auto-check when user completes them. User logs fruit, but "Log one fruit" stays unchecked.

**Step 1: Update TodaysJourney Component**

```tsx
interface TodaysJourneyProps {
  milestones: Array<{
    id: string;
    title: string;
    reward: number;
    completed: boolean; // This should come from backend
  }>;
  onTaskComplete?: (xpReward: number) => void;
}

export default function TodaysJourney({ milestones, onTaskComplete }: TodaysJourneyProps) {
  const completedCount = milestones.filter(m => m.completed).length;
  const allComplete = completedCount === milestones.length;
  const progress = (completedCount / milestones.length) * 100;

  // Previous completed count to detect new completions
  const [prevCompletedCount, setPrevCompletedCount] = useState(completedCount);

  // Trigger celebration when a new task is completed
  useEffect(() => {
    if (completedCount > prevCompletedCount && onTaskComplete) {
      const newlyCompletedTask = milestones.find((m, i) => 
        m.completed && !milestones[i]?.completed
      );
      if (newlyCompletedTask) {
        onTaskComplete(newlyCompletedTask.reward);
      }
    }
    setPrevCompletedCount(completedCount);
  }, [completedCount, prevCompletedCount, milestones, onTaskComplete]);

  // Trigger bonus when all tasks complete
  useEffect(() => {
    if (allComplete && completedCount === milestones.length && onTaskComplete) {
      // Give +50 XP bonus for completing all tasks
      setTimeout(() => onTaskComplete(50), 500);
    }
  }, [allComplete, completedCount, milestones.length, onTaskComplete]);

  return (
    <div className="bg-white border-2 border-gray-200 rounded-2xl p-6 space-y-4">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-xl font-bold text-gray-900">Today's Journey</h3>
        <div className="text-sm font-semibold text-gray-600">
          {completedCount}/{milestones.length} tasks
        </div>
      </div>

      {/* Animated Progress Bar */}
      <div className="h-3 bg-gray-100 rounded-full overflow-hidden relative">
        <div 
          className="h-full bg-gradient-to-r from-orange-400 via-orange-500 to-orange-600 transition-all duration-700 ease-out"
          style={{ width: `${progress}%` }}
        />
      </div>

      {/* Task List */}
      <div className="space-y-3">
        {milestones.map((milestone) => (
          <div 
            key={milestone.id}
            className={`
              flex items-center justify-between p-4 rounded-xl transition-all duration-300
              ${milestone.completed 
                ? 'bg-green-50 border-2 border-green-200 scale-[1.02]' 
                : 'bg-gray-50 border-2 border-gray-200'
              }
            `}
          >
            <div className="flex items-center space-x-3">
              {/* Animated Checkbox */}
              <div className={`
                w-7 h-7 rounded-full border-2 flex items-center justify-center 
                transition-all duration-300 transform
                ${milestone.completed 
                  ? 'bg-green-500 border-green-500 scale-110' 
                  : 'border-gray-300 bg-white'
                }
              `}>
                {milestone.completed && (
                  <svg 
                    className="w-5 h-5 text-white animate-scale-in" 
                    fill="none" 
                    viewBox="0 0 24 24" 
                    stroke="currentColor"
                  >
                    <path 
                      strokeLinecap="round" 
                      strokeLinejoin="round" 
                      strokeWidth={3} 
                      d="M5 13l4 4L19 7" 
                    />
                  </svg>
                )}
              </div>

              {/* Task Text */}
              <span className={`
                font-medium text-base transition-all duration-300
                ${milestone.completed 
                  ? 'text-green-800 line-through' 
                  : 'text-gray-800'
                }
              `}>
                {milestone.title}
              </span>
            </div>

            {/* XP Badge */}
            <div className={`
              px-3 py-1 rounded-full font-bold text-sm
              ${milestone.completed
                ? 'bg-green-200 text-green-800'
                : 'bg-orange-100 text-orange-600'
              }
            `}>
              +{milestone.reward} XP
            </div>
          </div>
        ))}
      </div>

      {/* All Complete Celebration */}
      {allComplete && (
        <div className="mt-4 p-4 bg-gradient-to-r from-orange-100 to-yellow-100 border-2 border-orange-300 rounded-xl text-center animate-bounce-in">
          <p className="text-lg font-bold text-orange-800">
            üéâ Daily Journey Complete! üéâ
```

```tsx
import { formatDistanceToNow } from 'date-fns';

interface Log {
  id: string;
  type: 'food' | 'activity';
  content?: string;
  emojis?: string[];
  durationMin?: number;
  xpAwarded?: number;
  feedback?: string;
  ts: string;
}

interface RecentLogsListProps {
  logs: Log[];
}

export default function RecentLogsList({ logs }: RecentLogsListProps) {
  const getTimeAgo = (timestamp: string) => {
    try {
      return formatDistanceToNow(new Date(timestamp), { addSuffix: true });
    } catch {
      return 'Recently';
    }
  };

    if (log.emojis && log.emojis.length > 0) {
      return log.emojis.join(' ');

      return log.content.substring(0, 50);
```

```tsx
interface XPProgressBarProps {
  xpToday: number;
  xpGoal: number;
}

function XPProgressBar({ xpToday, xpGoal }: XPProgressBarProps) {
  const progress = Math.min((xpToday / xpGoal) * 100, 100);
  const isGoalReached = xpToday >= xpGoal;

  return (
    <div className="bg-white border-2 border-gray-200 rounded-2xl p-6 space-y-3">
      {/* Header */}
      <div className="flex items-center justify-between">
        <h3 className="text-lg font-bold text-gray-900">Daily XP Goal</h3>
        <div className="flex items-center space-x-2">
          <span className="text-2xl font-black text-orange-600">
            {xpToday}
          </span>
          <span className="text-sm text-gray-500">/ {xpGoal} XP</span>
        </div>
      </div>

      {/* Progress Bar with Shimmer */}
      <div className="relative h-5 bg-gray-100 rounded-full overflow-hidden">
        <div 
          className={`
            h-full transition-all duration-700 ease-out relative
            ${isGoalReached 
              ? 'bg-gradient-to-r from-green-400 via-green-500 to-green-600' 
              : 'bg-gradient-to-r from-orange-400 via-orange-500 to-orange-600'
            }
          `}
          style={{ width: `${progress}%` }}
        >
          {/* Shimmer Effect */}
          <div className="absolute inset-0 bg-gradient-to-r from-transparent via-white to-transparent opacity-30 animate-shimmer" />
        </div>

        {/* Goal Marker (if not reached) */}
        {!isGoalReached && progress > 20 && (
          <div className="absolute top-0 right-0 h-full flex items-center pr-2">
            <span className="text-xs font-bold text-gray-400">üéØ</span>
          </div>
        )}
      </div>

      {/* Status Message */}
      <div className="text-center">
        {isGoalReached ? (
          <p className="text-sm font-bold text-green-700">
            üéâ Goal reached! You're crushing it today!
          </p>
        ) : xpToday > 0 ? (
          <p className="text-sm text-gray-600">
            {xpGoal - xpToday} XP to go ‚Ä¢ You're {Math.round(progress)}% there!
          </p>
        ) : (
          <p className="text-sm text-gray-600">
            Start logging to earn XP and reach your daily goal!
          </p>
        )}
      </div>
    </div>
  );
}
```

```tsx
interface BadgeWithProgress {
  code: string;
  name: string;
  emoji: string;
  earned: boolean;
  progress?: {
    current: number;
    total: number;
  };
}

function BadgeCard({ badge }: { badge: BadgeWithProgress }) {
  const progressPercent = badge.progress 
    ? (badge.progress.current / badge.progress.total) * 100 
    : 0;

  return (
    <div 
      className={`
        aspect-square flex flex-col items-center justify-center
        rounded-2xl border-2 p-3 transition-all duration-300
        ${badge.earned 
          ? 'bg-gradient-to-br from-yellow-100 to-orange-100 border-orange-300 shadow-md' 
          : 'bg-gray-50 border-gray-200 opacity-60'
        }
        hover:scale-105 cursor-pointer
      `}
    >
      {/* Badge Icon */}
      <div className="relative">
        <span className="text-3xl mb-1 block">
          {badge.earned ? badge.emoji : 'üîí'}
        </span>
        
        {/* Progress Ring for Locked Badges */}
        {!badge.earned && badge.progress && badge.progress.current > 0 && (
          <div className="absolute inset-0 flex items-center justify-center">
            <svg className="w-12 h-12 transform -rotate-90">
              <circle
                cx="24"
                cy="24"
                r="20"
                stroke="#FFB366"
                strokeWidth="2"
                fill="none"
                strokeDasharray={`${progressPercent * 1.26} 126`}
                className="transition-all duration-500"
              />
            </svg>
          </div>
        )}
      </div>

      {/* Badge Name */}
      <span className="text-xs font-semibold text-gray-700 text-center line-clamp-1 mb-1">
        {badge.name}
      </span>

      {/* Progress Text */}
      {!badge.earned && badge.progress && (
        <span className="text-xs font-bold text-orange-600">
          {badge.progress.current}/{badge.progress.total}
        </span>
      )}
      
      {badge.earned && (
        <span className="text-xs text-green-600 font-bold">‚úì Earned</span>
      )}
    </div>
  );
}
```
